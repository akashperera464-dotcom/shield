   206→/* StatCard — count-up animated counter that triggers on scroll-into-view  */
   207→/* ─────────────────────────────────────────────────────────────────────── */
   208→function StatCard({
   209→  stat,
   210→  delay,
   211→}: {
   212→  stat: { label: string; value: string; target: number; spark: number[]; color: string };
   213→  delay: number;
   214→}) {
   215→  const { ref, visible } = useScrollReveal<HTMLDivElement>({ threshold: 0.3 });
   216→  const count = useCountUp(stat.target, visible, 1800);
   217→  return (
   218→    <div
   219→      ref={ref}
   220→      className={`glass-card-hover p-5 reveal-hidden ${visible ? "reveal-visible" : ""}`}
   221→      style={{ transitionDelay: `${delay}s` }}
   222→    >
   223→      <div className="flex items-end justify-between">
   224→        <div>
   225→          <div className="text-3xl font-bold text-gradient-animated sm:text-4xl">
   226→            {count}{stat.value.replace(/[0-9]/g, "")}
   227→          </div>
   228→          <div className="mt-1 text-xs uppercase tracking-wider text-ink-400">{stat.label}</div>
   229→        </div>
   230→        <Sparkline data={stat.spark} color={stat.color} width={64} height={24} />
   231→      </div>
   232→    </div>
   233→  );
   234→}

   378→/* ─────────────────────────────────────────────────────────────────────── */
   379→/* FeedbackForm — visible to any visitor. Validates + writes to the store. */
   380→/* On success: shows a "pending approval" confirmation.                    */
   381→/* ─────────────────────────────────────────────────────────────────────── */
   382→function FeedbackForm({ onDone }: { onDone: () => void }) {
   383→  const [form, setForm] = useState({ name: "", role: "", quote: "", rating: 5 });
   384→  const [submitting, setSubmitting] = useState(false);
   385→  const [error, setError] = useState<string | null>(null);
   386→  const [done, setDone] = useState(false);
   387→
   388→  const update = (k: keyof typeof form, v: string | number) =>
   389→    setForm((f) => ({ ...f, [k]: v }));
   390→
   391→  const validate = (): string | null => {
   392→    if (!form.name.trim()) return "Please enter your name.";
   393→    if (!form.role.trim()) return "Please enter your role / company.";
   394→    if (form.quote.trim().length < 10) return "Tell us a bit more — at least 10 characters.";
   395→    if (form.quote.trim().length > 500) return "Keep it under 500 characters.";
   396→    if (form.rating < 1 || form.rating > 5) return "Pick a rating between 1 and 5 stars.";
   397→    return null;
   398→  };
   399→
   400→  const handleSubmit = async (e: React.FormEvent) => {
   401→    e.preventDefault();
   402→    const err = validate();
   403→    if (err) {
   404→      setError(err);
   405→      return;
   406→    }
   407→    setError(null);
   408→    setSubmitting(true);
   409→    // Simulate async to make the success state feel intentional
   410→    await new Promise((r) => setTimeout(r, 350));
   411→    addFeedback({
   412→      name: form.name,
   413→      role: form.role,
   414→      rating: form.rating,
   415→      quote: form.quote,
   416→    });
   417→    setSubmitting(false);
   418→    setDone(true);
   419→    // Auto-close after 4s
   420→    setTimeout(() => {
   421→      setDone(false);
   422→      setForm({ name: "", role: "", quote: "", rating: 5 });
   423→      onDone();
   424→    }, 4000);
   425→  };
   426→
   427→  if (done) {
   428→    return (
   429→      <Reveal className="glass-card-gradient mx-auto max-w-2xl p-8 text-center">
   430→        <div className="mx-auto mb-3 flex h-12 w-12 items-center justify-center rounded-full bg-emerald-500/15">
   431→          <CheckCircle2 className="h-6 w-6 text-emerald-300" />
   432→        </div>
   433→        <h3 className="text-lg font-bold text-white">Thank you for your feedback!</h3>
   434→        <p className="mt-2 text-sm text-ink-300">
   435→          Your testimonial has been saved and is now awaiting approval from our team.
   436→          Once approved, it will appear here for the world to see.
   437→        </p>
   438→      </Reveal>
   439→    );
   440→  }
   441→
   442→  return (
   443→    <Reveal className="glass-card mx-auto max-w-2xl p-6 sm:p-8">
   444→      <div className="mb-5 flex items-center gap-2">
   445→        <MessageSquare className="h-5 w-5 text-mint-300" />
   446→        <h3 className="text-lg font-bold text-white">Share your experience</h3>
   447→      </div>
   448→      <p className="mb-5 text-sm text-ink-400">
   449→        Worked with The Shield? Tell other founders what it was like. Your feedback will be
   450→        reviewed by our team before appearing publicly.
   451→      </p>
   452→
   453→      <form onSubmit={handleSubmit} className="space-y-4">
   454→        <div className="grid gap-4 sm:grid-cols-2">
   455→          <div>
   456→            <label className="mb-1.5 block text-xs font-medium text-ink-300">
   457→              Your name <span className="text-rose-400">*</span>
   458→            </label>
   459→            <input
   460→              type="text"
   461→              value={form.name}
   462→              onChange={(e) => update("name", e.target.value)}
   463→              placeholder="e.g. Sara Al-Mansoori"
   464→              className="input-field"
   465→              maxLength={80}
   466→              required
   467→            />
   468→          </div>
   469→          <div>
   470→            <label className="mb-1.5 block text-xs font-medium text-ink-300">
   471→              Role / Company <span className="text-rose-400">*</span>
   472→            </label>
   473→            <input
   474→              type="text"
   475→              value={form.role}
   476→              onChange={(e) => update("role", e.target.value)}
   477→              placeholder="e.g. CEO, Layla Cosmetics"
   478→              className="input-field"
   479→              maxLength={80}
   480→              required
   481→            />
   482→          </div>
   483→        </div>
   484→
   485→        <div>
   486→          <label className="mb-1.5 block text-xs font-medium text-ink-300">
   487→            Your rating <span className="text-rose-400">*</span>
   488→          </label>
   489→          <div className="flex items-center gap-1.5">
   490→            {[1, 2, 3, 4, 5].map((n) => (
   491→              <button
   492→                key={n}
   493→                type="button"
   494→                onClick={() => update("rating", n)}
   495→                className="rounded-md p-1 transition-transform hover:scale-110"
   496→                aria-label={`${n} star${n > 1 ? "s" : ""}`}
   497→              >
   498→                <Star
   499→                  className={`h-7 w-7 transition-colors ${
   500→                    n <= form.rating

   526→        </div>
   527→
   528→        {error && (
   529→          <div className="rounded-md border border-rose-500/30 bg-rose-500/10 px-3 py-2 text-xs text-rose-300">
   530→            {error}
   531→          </div>
   532→        )}
   533→
   534→        <div className="flex items-center justify-between gap-3 pt-2">
   535→          <span className="inline-flex items-center gap-1.5 text-[10px] uppercase tracking-wider text-ink-500">
   536→            <Clock className="h-3 w-3" /> Reviewed by our team before publishing
   537→          </span>
   538→          <button
   539→            type="submit"
   540→            disabled={submitting}
   541→            className="btn-primary shine-sweep text-sm"

   544→              <>
   545→                <Loader2 className="h-4 w-4 animate-spin" /> Submitting…
